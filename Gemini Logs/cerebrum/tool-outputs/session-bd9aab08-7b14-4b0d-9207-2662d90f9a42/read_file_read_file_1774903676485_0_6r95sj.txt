Evaluate at each incompleteness level\n    # ------------------------------------------------------------------\n    results: List[Dict] = []\n    for level in sorted(args.levels):\n        removal_fraction = INCOMPLETENESS_LEVELS.get(level, 0.0)\n        label = LEVEL_LABELS.get(level, f\"Level {level}\")\n        print(f\"  Level {level} — {label}\")\n        print(f\"    Removing {removal_fraction*100:.0f}% of answer-adjacent edges...\")\n\n        G_inc = apply_incompleteness(G_full, qa_all, removal_fraction, rng)\n        edges_removed = G_full.number_of_edges() - G_inc.number_of_edges()\n        print(f\"    Graph: {G_inc.number_of_edges():,} edges \"\n              f\"({edges_removed:,} removed)\")\n\n        r = evaluate_level(\n            G_inc, qa_all, embeddings, cmap_full,\n            sentence_engine, question_embeddings,\n            beam_width=args.beam_width,\n            top_k=args.top_k,\n            coarsen_target=args.coarsen_target,\n            use_rem=args.rem,\n            level=level,\n            removal_fraction=removal_fraction,\n        )\n        results.append(r)\n        print(f\"    Hits@1={r['hits_1']:.4f}  Hits@10={r['hits_10']:.4f}  \"\n              f\"MRR={r['mrr']:.4f}  \"\n              f\"Conf={r['mean_confidence']:.3f}  \"\n              f\"({r['ms_per_q']:.2f}ms/Q)\")\n        print()\n\n    # ------------------------------------------------------------------\n    # 6. Results table\n    # ------------------------------------------------------------------\n    print(\"=\" * 72)\n    print(\"  IKGWQ Results — Graceful Degradation Profile\")\n    print(\"=\" * 72)\n    print()\n    print(f\"  {'Level':<18}  {'Remove%':>7}  {'Hits@1':>8}  \"\n          f\"{'Hits@10':>8}  {'MRR':>8}  {'Conf':>6}  {'ms/Q':>7}\")\n    print(f\"  {'-'*18}  {'-'*7}  {'-'*8}  {'-'*8}  {'-'*8}  {'-'*6}  {'-'*7}\")\n\n    baseline_h1  = results[0][\"hits_1\"]  if results else 0\n    baseline_h10 = results[0][\"hits_10\"] if results else 0\n    for r in results:\n        lvl   = LEVEL_LABELS.get(r[\"level\"], f\"Level {r['level']}\")\n        delta = r[\"hits_1\"] - baseline_h1\n        sign  = \"+\" if delta >= 0 else \"\"\n        print(f\"  {lvl:<18}  {r['removal_pct']:>6.0f}%  \"\n              f\"{r['hits_1']:>8.4f}  {r['hits_10']:>8.4f}  \"\n              f\"{r['mrr']:>8.4f}  {r['mean_confidence']:>6.3f}  \"\n              f\"{r['ms_per_q']:>6.2f}  ({sign}{delta:+.4f})\")\n    print()\n\n    # Graceful degradation score = AUC over incompleteness curve\n    if len(results) > 1:\n        h1_vals = [r[\"hits_1\"]  for r in results]\n        h10_vals = [r[\"hits_10\"] for r in results]\n        # Relative AUC: area under normalised degradation curve\n        # 1.0 = no degradation; 0.0 = complete failure at first removal\n        rel_auc_h1  = sum(h1_vals)  / (len(h1_vals)  * max(h1_vals[0],  1e-9))\n        rel_auc_h10 = sum(h10_vals) / (len(h10_vals) * max(h10_vals[0], 1e-9))\n        print(f\"  Graceful Degradation Score (relative AUC):\")\n        print(f\"    Hits@1  AUC = {rel_auc_h1:.4f}  \"\n              f\"(1.0 = perfect retention under incompleteness)\")\n        print(f\"    Hits@10 AUC = {rel_auc_h10:.4f}\")\n        print()\n\n    print(\"  Key insight:\")\n    print(\"  CEREBRUM's path_confidence score tracks accuracy degradation.\")\n    print(\"  Lower confidence = model knows it is less certain — honest uncertainty.\")\n    print(\"  LLM-augmented systems maintain artificially high scores by using\")\n    print(\"  memorised facts to bypass missing edges — this is not KG reasoning.\")\n    print()\n\n    if args.rem:\n        print(\"  REM Engine was active: synthesised edges partially compensate\")\n        print(\"  for removed edges using graph-native inference (no LLM).\")\n        print()\n\n    # ------------------------------------------------------------------\n    # 7. Save results\n    # ------------------------------------------------------------------\n    import csv\n    out = CACHE_DIR / \"ikgwq_results.csv\"\n    if results:\n        with open(out, \"w\", newline=\"\") as f:\n            writer = csv.DictWriter(f, fieldnames=results[0].keys())\n            writer.writeheader()\n            writer.writerows(results)\n        print(f\"  Results saved to {out}\")\n\n\nif __name__ == \"__main__\":\n    main()\n"
